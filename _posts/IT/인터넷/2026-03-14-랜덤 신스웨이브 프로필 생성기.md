    <div style="display: grid; grid-template-columns: 1fr 1fr 1fr; gap: 5px; margin-bottom: 15px;">
        <div>
            <input type="color" id="colorSunTop" value="#ff00ff" oninput="updateColors()" style="width:100%; height:30px; border:none; background:none; cursor:pointer;">
        </div>
        <div>
            <input type="color" id="colorSunBottom" value="#ffcc00" oninput="updateColors()" style="width:100%; height:30px; border:none; background:none; cursor:pointer;">
        </div>
        <div>
            <input type="color" id="colorGrid" value="#00ffff" oninput="updateColors()" style="width:100%; height:30px; border:none; background:none; cursor:pointer;">
        </div>
    </div>

    <div style="display: flex; flex-direction: column; gap: 8px;">
        <div style="display: flex; gap: 5px;">
            <button onclick="randomize()" style="flex: 1; background: #ff00ff; border: none; color: white; padding: 10px; font-weight: bold; cursor: pointer; border-radius: 5px; font-size: 11px;">RANDOM</button>
            <button onclick="downloadPng()" style="flex: 2; background: #00ffff; border: none; color: black; padding: 10px; font-weight: bold; cursor: pointer; border-radius: 5px; font-size: 11px;">DOWNLOAD PNG</button>
        </div>
    </div>

    <script>
        function updateColors() {
            const top = document.getElementById('colorSunTop').value;
            const bottom = document.getElementById('colorSunBottom').value;
            const grid = document.getElementById('colorGrid').value;

            document.getElementById('stop-top').setAttribute('stop-color', top);
            document.getElementById('stop-bottom').setAttribute('stop-color', bottom);
            document.getElementById('grid-group').setAttribute('stroke', grid);
        }

        function randomize() {
            const r = () => '#' + Math.floor(Math.random()*16777215).toString(16).padStart(6, '0');
            document.getElementById('colorSunTop').value = r();
            document.getElementById('colorSunBottom').value = r();
            document.getElementById('colorGrid').value = r();
            updateColors();
        }

        function downloadPng() {
            const svg = document.getElementById('synth-svg');
            const serializer = new XMLSerializer();
            const svgString = serializer.serializeToString(svg);
            const svgBlob = new Blob([svgString], {type: 'image/svg+xml;charset=utf-8'});
            const url = URL.createObjectURL(svgBlob);

            const canvas = document.createElement('canvas');
            canvas.width = 250;
            canvas.height = 250;
            const ctx = canvas.getContext('2d');

            const img = new Image();
            img.onload = function() {
                ctx.clearRect(0, 0, 250, 250);
                ctx.drawImage(img, 0, 0);
                URL.revokeObjectURL(url);
                const pngUrl = canvas.toDataURL("image/png");
                const link = document.createElement('a');
                link.download = `synthwave-profile.png`;
                link.href = pngUrl;
                link.click();
            };
            img.src = url;
        }
    </script>
</div>
