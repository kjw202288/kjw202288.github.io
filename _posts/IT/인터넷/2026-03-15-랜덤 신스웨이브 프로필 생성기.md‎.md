    <script>
        function update() {
            const sunTop = document.getElementById('inputSunTop').value;
            const sunBottom = document.getElementById('inputSunBottom').value;
            const bgTop = document.getElementById('inputBGTop').value;
            const bgBottom = document.getElementById('inputBGBottom').value;
            const stars = document.getElementById('inputStars').value;

            document.getElementById('sun-top').setAttribute('stop-color', sunTop);
            document.getElementById('sun-bottom').setAttribute('stop-color', sunBottom);
            document.getElementById('ref-top').setAttribute('stop-color', sunBottom);
            document.getElementById('ref-bottom').setAttribute('stop-color', sunBottom);
            document.getElementById('bg-top').setAttribute('stop-color', bgTop);
            document.getElementById('bg-bottom').setAttribute('stop-color', bgBottom);
            document.getElementById('star-group').setAttribute('fill', stars);
        }

        function updateStars() {
            const starCnt = document.getElementById('inputStarCnt').value;
            const starGroup = document.getElementById('star-group');
            starGroup.innerHTML = '';
            for (let i = 0; i < starCnt; i++) {
                const circle = document.createElementNS('http://www.w3.org/2000/svg', 'circle');
                circle.setAttribute('cx', Math.random() * 300);
                circle.setAttribute('cy', Math.random() * 300);
                circle.setAttribute('r', Math.random() * 0.9 + 0.1);
                starGroup.appendChild(circle);
            }
        }

        function randomize() {
            const hex = () => '#' + Math.floor(Math.random()*16777215).toString(16).padStart(6, '0');
            if (!document.getElementById('lockSunTop').checked) document.getElementById('inputSunTop').value = hex();
            if (!document.getElementById('lockSunBottom').checked) document.getElementById('inputSunBottom').value = hex();
            if (!document.getElementById('lockBGTop').checked) document.getElementById('inputBGTop').value = hex();
            if (!document.getElementById('lockBGBottom').checked) document.getElementById('inputBGBottom').value = hex();
            if (!document.getElementById('lockStars').checked) document.getElementById('inputStars').value = hex();
            document.getElementById('inputStarCnt').value = Math.floor(Math.random() * 120) + 20;
            update();
            updateStars();
        }

        function download() {
            const svg = document.getElementById('synth-svg');
            const size = parseInt(document.getElementById('saveSize').value) || 1000;
            const format = document.getElementById('saveFormat').value;
            const extension = format.split('/')[1].replace('x-icon', 'ico').replace('jpeg', 'jpg');

            const svgData = new XMLSerializer().serializeToString(svg);
            const canvas = document.createElement('canvas');
            const ctx = canvas.getContext('2d');
            const img = new Image();
            
            canvas.width = size;
            canvas.height = size;
            
            const svgBlob = new Blob([svgData], {type: 'image/svg+xml;charset=utf-8'});
            const url = URL.createObjectURL(svgBlob);

            img.onload = function() {
                // JPG일 경우 배경색 채우기 (투명도 미지원 대응)
                if (format === 'image/jpeg') {
                    ctx.fillStyle = document.getElementById('inputBGBottom').value;
                    ctx.fillRect(0, 0, size, size);
                }
                ctx.drawImage(img, 0, 0, size, size);
                URL.revokeObjectURL(url);
                
                const link = document.createElement('a');
                link.download = `synthwave-avatar.${extension}`;
                link.href = canvas.toDataURL(format, 0.9);
                link.click();
            };
            img.src = url;
        }

        updateStars();
    </script>
