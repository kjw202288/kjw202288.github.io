    <script>
        const $ = id => document.getElementById(id);
        const u = () => {
            const r1Color = $('c2').value;
            $('st1').setAttribute('stop-color', $('c1').value);
            $('st2').setAttribute('stop-color', r1Color);
            ['rt','rb'].forEach(id=>$(id).setAttribute('stop-color', r1Color));
            $('bt').setAttribute('stop-color', $('c3').value);
            $('bb').setAttribute('stop-color', $('c4').value);
            $('sg').setAttribute('fill', $('cs').value);

            // 1. 태양 크기 업데이트
            const sunR = parseFloat($('ssz').value);
            $('sun-circle').setAttribute('r', sunR);

            // 2. 태양 크기에 비례하여 하단 반사광(rl) 크기 및 위치 동적 계산
            // 기준 반지름 130일 때의 값들을 기준으로 비율 계산
            const ratio = sunR / 130; 
            const baseWidths = [260, 230, 180, 120]; // r=130일 때의 기본 너비들
            const center = 150; // SVG 중앙 x좌표

            baseWidths.forEach((baseWidth, index) => {
                const rectId = `r${index + 1}`;
                const newWidth = baseWidth * ratio;
                const newX = center - (newWidth / 2); // 중앙 정렬을 위한 x 좌표 계산
                
                $(rectId).setAttribute('width', newWidth);
                $(rectId).setAttribute('x', newX);
            });
        };
        const us = () => {
            $('sg').innerHTML = '';
            for(let i=0; i<$('sc').value; i++) {
                const c = document.createElementNS('http://www.w3.org/2000/svg','circle');
                c.setAttribute('cx', Math.random()*300); c.setAttribute('cy', Math.random()*300); c.setAttribute('r', Math.random()*0.9+0.1);
                $('sg').appendChild(c);
            }
        };
        const rz = () => {
            const h = () => '#'+Math.random().toString(16).slice(2,8).padEnd(6,'0');
            ['1','2','3','4','s'].forEach(id => { if(!$('l'+id).checked) $('c'+(id==='s'?'s':id)).value = h(); });
            $('sc').value = Math.floor(Math.random()*120)+20; u(); us();
        };
        const dl = () => {
            const v = $('s'), can = document.createElement('canvas'), ctx = can.getContext('2d'), img = new Image();
            const s = parseInt($('sz').value)||1000, f = $('sf').value, xml = new XMLSerializer().serializeToString(v);
            const ext = f.includes('x-icon') ? 'ico' : f.split('/')[1].replace('jpeg','jpg');
            can.width = can.height = s;
            img.onload = () => {
                if(f.includes('jpeg')) { ctx.fillStyle=$('c4').value; ctx.fillRect(0,0,s,s); }
                ctx.drawImage(img, 0, 0, s, s);
                const a = document.createElement('a');
                a.download = `zoom-sun-avatar.${ext}`; a.href = can.toDataURL(f, 0.9); a.click();
            };
            img.src = 'data:image/svg+xml;base64,' + btoa(unescape(encodeURIComponent(xml)));
        };
        u(); us();
    </script>
</div>
